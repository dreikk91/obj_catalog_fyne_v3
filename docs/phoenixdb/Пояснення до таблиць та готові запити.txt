SELECT 
	DISTINCT RealPanel.panel_id, 
	groups.group_, 
	groups.message, 
	Company.Address AS Address, 
	engineers.engineer_name, 
	CASE 
		WHEN  
Mphone.RadioType <> 16 THEN MphoneRadioType.Message 
	END AS MphoneRadioTypeMessage, 
	CASE 
		WHEN Mphone.RadioType = 16 THEN MphoneRadioType.Message 
	END AS MphoneRadioTypeMessage1, 
	CASE 
		WHEN Mphone.RadioType <> 16 THEN MphoneTypeTransmitter.Message 
	END AS MphoneTypeTransmitterMessage, 
	TypeDevices.TypeDevice AS CentralName, 
	Company.Director, 
	Company.CompanyName, 
	Masters.MasterName, 
	RealPanel.TestPanel, 
	ISDisabled = CASE 
		WHEN RealPanel.disabled = 1 THEN CAST(1 AS bit) 
		WHEN groups.disabled = 1 THEN CAST(1 AS bit) 
		ELSE CAST (0 AS bit) 
	END, 
	RealPanel.CreateDate, 
	RealPanel.Movable_Object, 
	RealPanel.Panel_Type, 
	LEN(RealPanel.panel_id) AS PanelIdLen, 
	CompanyType.Name AS CompanyType, 
	RealPanel.FreeArrivalsCount, 
	Groups.isOpen AS [status], 
	Groups.TimeEvent AS status_time 
FROM 
	vwRealPanel AS RealPanel WITH (NOLOCK) 
LEFT OUTER JOIN Groups WITH(NOLOCK) ON 
	RealPanel.Realpanel_id = groups.panel_id 
LEFT OUTER JOIN company WITH(NOLOCK) ON 
	groups.companyID = company.ID 
LEFT OUTER JOIN CompanyType WITH(NOLOCK) ON 
	CompanyType.ID = company.TypeID 
LEFT OUTER JOIN engineers WITH(NOLOCK) ON 
	engineers.engineer_id = RealPanel.engineer_id 
LEFT OUTER JOIN Masters WITH(NOLOCK) ON 
	Masters.Master_id = RealPanel.Master_id 
LEFT OUTER JOIN Mphone WITH(NOLOCK) ON 
	Mphone.panel_id = RealPanel.Realpanel_id 
LEFT OUTER JOIN MphoneRadioType WITH(NOLOCK) ON 
	MphoneRadioType.RadioType_id = Mphone.RadioType 
LEFT OUTER JOIN MphoneTypeTransmitter WITH(NOLOCK) ON 
	MphoneTypeTransmitter.TypeTransmitter_id = MPhone.TypeTransmitter 
LEFT OUTER JOIN Central WITH(NOLOCK) ON 
	Central.Panel_id = RealPanel.Realpanel_id 
LEFT OUTER JOIN TypeDevices WITH(NOLOCK) ON 
	TypeDevices.Device_id = central.Device_id 
LEFT OUTER JOIN Pults WITH(NOLOCK) ON 
	Pults.Id = RealPanel.Pult_id 
WHERE 
	COALESCE(Pults.IsOutPult, 0) = 0 
ORDER BY 
	RealPanel.panel_id, 
	groups.group_ 
Кожен об'єкт може мати декілька груп, з різними назвами
panel_id номер об'єкту
group_	номер групи
message	назва групи
Address	адрес
engineer_name	ім'я інженера обслуговування (розширена інформація)
MphoneRadioTypeMessage	тип приладу
MphoneRadioTypeMessage1	не використовується
MphoneTypeTransmitterMessage	назва радіомодулю (додати до назви приладу)
CentralName	не використовується
Director	група реагування (розширена інформація)
CompanyName	Назва об'єкту
MasterName 	(розширена інформація)
TestPanel чи заблоковано об'єкт, 1 так 2 ні
ISDisabled	чи заблоковано об'єкт, 1 так 2 ні
CreateDate дата створення
Movable_Object	не використовується
Panel_Type	1 якщо пожежний об'єкт, 0 охоронний
PanelIdLen не використовується
CompanyType	тип об'єкту
FreeArrivalsCount	не використовується
status	1 або 0
status_time час останньої події

SELECT
	panel.*,
	Engineers.engineer_name,
	Masters.MasterName,
	Installers.InstallerName,
	CONVERT(VARCHAR(10), CreateDate, 104) AS CD,
	central.panel_id AS central_id,
	mphone.panel_id AS mphone,
	stands.panel_id AS stand ,
	ISNULL(Groups.group_,-1) AS group_,
	Areas.Area_name,
	Pults.*,
	COALESCE(MPhone.LastTest, Central.LastTest) AS LastTest
FROM
	Panel WITH (NOLOCK)
LEFT OUTER JOIN Engineers WITH (NOLOCK) ON
	panel.engineer_id = engineers.engineer_id
LEFT OUTER JOIN Masters WITH (NOLOCK) ON
	panel.master_id = Masters.master_id
LEFT OUTER JOIN Installers WITH (NOLOCK) ON
	panel.installer_id = Installers.installer_id
LEFT JOIN Central WITH (NOLOCK) ON
	Central.panel_id = Panel.panel_id
LEFT OUTER JOIN groups WITH (NOLOCK) ON
	Panel.panel_id = Groups.panel_id
LEFT OUTER JOIN mphone WITH (NOLOCK) ON
	Panel.panel_id = mphone.panel_id
LEFT OUTER JOIN stands WITH (NOLOCK) ON
	((Panel.panel_id = stands.panel_id)
		AND(Stands.standorkey = 0))
LEFT OUTER JOIN Areas WITH (NOLOCK) ON
	Panel.Area_id = Areas.Area_id
LEFT OUTER JOIN Pults WITH (NOLOCK) ON
	Panel.Pult_id = Pults.id
WHERE
	panel.panel_id = 'L00001

SELECT
	a.*,
	b.CompanyName,
	b.address,
	Telephones = b.Telephones,
	b.Director,
	b.Director2,
	b.Memo,
	b.TypeID,
	ct.Name AS TypeName,
	MoreAboutLun7.info_id,
	MoreAboutLun7.version,
	MoreAboutLun7.ExType_id,
	ExtensionTypes.Name AS ExtensionName,
	CONVERT(VARCHAR(5), Z8Timeout, 108) AS z8,
	CONVERT(VARCHAR(5), TimeForClose, 108) AS tfc ,
	engineers.engineer_name ,
	Customers.Organization,
	Customers.OrganizationAddress,
	Customers.Director AS CustomerDirector,
	ServiceOrganization.ServiceName,
	ISNULL(b.ServiceID,-1) AS ServiceID ,
	ISNULL(b.CustomerID,-1) AS CustomerID,
	NeedSendMessage = CASE
		WHEN EXISTS(SELECT * FROM dbo.Group_SMS_Code WHERE dbo.Group_SMS_Code.Panel_id = a.Panel_id AND dbo.Group_SMS_Code.Group_ = a.Group_) THEN 1
		ELSE 0
	END
FROM
	groups AS a WITH (NOLOCK)
INNER JOIN company AS b WITH (NOLOCK) ON
	a.CompanyID = b.ID
LEFT JOIN CompanyType AS ct WITH (NOLOCK) ON
	ct.ID = b.TypeID
LEFT JOIN MoreAboutLun7 WITH (NOLOCK) ON
	a.info_id = MoreAboutLun7.info_id
LEFT JOIN ExtensionTypes WITH (NOLOCK) ON
	MoreAboutLun7.ExType_id = ExtensionTypes.ExType_id
LEFT JOIN engineers WITH (NOLOCK) ON
	engineers.Work_Panel_id = a.Panel_id
LEFT JOIN vwRealPanel WITH (NOLOCK) ON
	vwRealPanel.Panel_id = a.panel_id
LEFT JOIN Pults WITH (NOLOCK) ON
	Pults.Id = vwRealPanel.Pult_Id
LEFT OUTER JOIN dbo.Customers ON
	b.CustomerID = dbo.Customers.ID
LEFT OUTER JOIN dbo.ServiceOrganization ON
	b.ServiceID = dbo.ServiceOrganization.ID
WHERE
	a.panel_id = 'L00028'
	AND group_ = 0

SELECT
	G.Panel_id AS panel_id,
	G.Group_ AS group_no,
	G.Message AS group_name,
	G.IsOpen AS is_open,
	G.TimeEvent AS group_time_event,
	G.disabled AS group_disabled,
	C.CompanyName AS company_name,
	C.Address AS company_address,
	COALESCE(PrimaryPhone.phone_number, C.Telephones) AS telephones,
	CAST(NULL AS nvarchar(255)) AS type_name,
	P.Disabled AS panel_disabled,
	P.TestPanel AS test_panel,
	P.Panel_type AS panel_type,
	T.StateEvent AS state_event,
	P.CreateDate AS create_date,
	P.DateLastChange AS date_last_change,
	CAST(NULL AS nvarchar(255)) AS engineer_name,
	PrimarySIM.sim_number AS sim1_number,
	SecondarySIM.sim_number AS sim2_number
FROM Groups G WITH (NOLOCK)
LEFT JOIN Company C WITH (NOLOCK) ON C.ID = G.CompanyID
LEFT JOIN (
	SELECT Panel_id, Group_, MAX(StateEvent) AS StateEvent
	FROM Temp WITH (NOLOCK)
	GROUP BY Panel_id, Group_
) T ON T.Panel_id = G.Panel_id AND T.Group_ = G.Group_
INNER JOIN vwRealPanel P WITH (NOLOCK) ON P.Panel_id = G.Panel_id
OUTER APPLY (
	SELECT TOP (1)
		LTRIM(RTRIM(ISNULL(RT.PhoneNo, ''))) AS phone_number
	FROM Responsibles R WITH (NOLOCK)
	INNER JOIN ResponsibleTel RT WITH (NOLOCK) ON RT.ResponsiblesList_id = R.ResponsiblesList_id
	LEFT JOIN ResponsibleTelDescription RTD WITH (NOLOCK) ON
		RTD.Responsible_id = R.Responsible_id
		AND RTD.ResponsibleTel_id = RT.ResponsibleTel_id
	WHERE
		R.panel_id = G.Panel_id
		AND LTRIM(RTRIM(ISNULL(RT.PhoneNo, ''))) <> ''
	ORDER BY
		ISNULL(RTD.CallOrder, R.Responsible_Number),
		R.Responsible_Number,
		RT.ResponsibleTel_id
) AS PrimaryPhone
OUTER APPLY (
	SELECT TOP (1)
		COALESCE(NULLIF(LTRIM(RTRIM(S.SimNumber)), ''), NULLIF(LTRIM(RTRIM(S.RealSimNumber)), '')) AS sim_number
	FROM Mphone M WITH (NOLOCK)
	INNER JOIN OnBoardDevice D WITH (NOLOCK) ON D.Mphone_id = M.Mphone_id
	INNER JOIN OnBoardDeviceType DT WITH (NOLOCK) ON DT.OnBoardDeviceType_ID = D.OnBoardDeviceType_ID
	INNER JOIN Sim S WITH (NOLOCK) ON S.OnBoardDevice_ID = D.OnBoardDevice_ID
	WHERE
		M.Panel_id = G.Panel_id
		AND DT.Enum = 0
		AND COALESCE(NULLIF(LTRIM(RTRIM(S.SimNumber)), ''), NULLIF(LTRIM(RTRIM(S.RealSimNumber)), '')) IS NOT NULL
	ORDER BY
		CASE WHEN ISNULL(S.IsCurrentSim, 0) = 1 THEN 0 ELSE 1 END,
		CASE WHEN ISNULL(S.IsMainSim, 0) = 1 THEN 0 ELSE 1 END,
		S.Sim_ID
) AS PrimarySIM
OUTER APPLY (
	SELECT TOP (1)
		COALESCE(NULLIF(LTRIM(RTRIM(S.SimNumber)), ''), NULLIF(LTRIM(RTRIM(S.RealSimNumber)), '')) AS sim_number
	FROM Mphone M WITH (NOLOCK)
	INNER JOIN OnBoardDevice D WITH (NOLOCK) ON D.Mphone_id = M.Mphone_id
	INNER JOIN OnBoardDeviceType DT WITH (NOLOCK) ON DT.OnBoardDeviceType_ID = D.OnBoardDeviceType_ID
	INNER JOIN Sim S WITH (NOLOCK) ON S.OnBoardDevice_ID = D.OnBoardDevice_ID
	WHERE
		M.Panel_id = G.Panel_id
		AND DT.Enum = 1
		AND COALESCE(NULLIF(LTRIM(RTRIM(S.SimNumber)), ''), NULLIF(LTRIM(RTRIM(S.RealSimNumber)), '')) IS NOT NULL
	ORDER BY
		CASE WHEN ISNULL(S.IsCurrentSim, 0) = 1 THEN 0 ELSE 1 END,
		CASE WHEN ISNULL(S.IsMainSim, 0) = 1 THEN 0 ELSE 1 END,
		S.Sim_ID
) AS SecondarySIM
ORDER BY G.Panel_id, G.Group_

				CASE WHEN ISNULL(S.IsCurrentSim, 0) = 1 THEN 0 ELSE 1 END,
				CASE WHEN ISNULL(S.IsMainSim, 0) = 1 THEN 0 ELSE 1 END,
				S.Sim_ID
		) AS rn
	FROM Mphone M WITH (NOLOCK)
	INNER JOIN OnBoardDevice D WITH (NOLOCK) ON D.Mphone_id = M.Mphone_id
	INNER JOIN OnBoardDeviceType DT WITH (NOLOCK) ON DT.OnBoardDeviceType_ID = D.OnBoardDeviceType_ID
	INNER JOIN Sim S WITH (NOLOCK) ON S.OnBoardDevice_ID = D.OnBoardDevice_ID
	WHERE COALESCE(NULLIF(LTRIM(RTRIM(S.SimNumber)), ''), NULLIF(LTRIM(RTRIM(S.RealSimNumber)), '')) IS NOT NULL
)
SELECT
	panel_id,
	MAX(CASE WHEN sim_slot = 0 AND rn = 1 THEN sim_number END) AS sim1_number,
	MAX(CASE WHEN sim_slot = 1 AND rn = 1 THEN sim_number END) AS sim2_number
FROM SimBase
GROUP BY panel_id
