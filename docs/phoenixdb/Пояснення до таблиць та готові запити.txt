SELECT 
	DISTINCT RealPanel.panel_id, 
	groups.group_, 
	groups.message, 
	Company.Address AS Address, 
	engineers.engineer_name, 
	CASE 
		WHEN  
Mphone.RadioType <> 16 THEN MphoneRadioType.Message 
	END AS MphoneRadioTypeMessage, 
	CASE 
		WHEN Mphone.RadioType = 16 THEN MphoneRadioType.Message 
	END AS MphoneRadioTypeMessage1, 
	CASE 
		WHEN Mphone.RadioType <> 16 THEN MphoneTypeTransmitter.Message 
	END AS MphoneTypeTransmitterMessage, 
	TypeDevices.TypeDevice AS CentralName, 
	Company.Director, 
	Company.CompanyName, 
	Masters.MasterName, 
	RealPanel.TestPanel, 
	ISDisabled = CASE 
		WHEN RealPanel.disabled = 1 THEN CAST(1 AS bit) 
		WHEN groups.disabled = 1 THEN CAST(1 AS bit) 
		ELSE CAST (0 AS bit) 
	END, 
	RealPanel.CreateDate, 
	RealPanel.Movable_Object, 
	RealPanel.Panel_Type, 
	LEN(RealPanel.panel_id) AS PanelIdLen, 
	CompanyType.Name AS CompanyType, 
	RealPanel.FreeArrivalsCount, 
	Groups.isOpen AS [status], 
	Groups.TimeEvent AS status_time 
FROM 
	vwRealPanel AS RealPanel WITH (NOLOCK) 
LEFT OUTER JOIN Groups WITH(NOLOCK) ON 
	RealPanel.Realpanel_id = groups.panel_id 
LEFT OUTER JOIN company WITH(NOLOCK) ON 
	groups.companyID = company.ID 
LEFT OUTER JOIN CompanyType WITH(NOLOCK) ON 
	CompanyType.ID = company.TypeID 
LEFT OUTER JOIN engineers WITH(NOLOCK) ON 
	engineers.engineer_id = RealPanel.engineer_id 
LEFT OUTER JOIN Masters WITH(NOLOCK) ON 
	Masters.Master_id = RealPanel.Master_id 
LEFT OUTER JOIN Mphone WITH(NOLOCK) ON 
	Mphone.panel_id = RealPanel.Realpanel_id 
LEFT OUTER JOIN MphoneRadioType WITH(NOLOCK) ON 
	MphoneRadioType.RadioType_id = Mphone.RadioType 
LEFT OUTER JOIN MphoneTypeTransmitter WITH(NOLOCK) ON 
	MphoneTypeTransmitter.TypeTransmitter_id = MPhone.TypeTransmitter 
LEFT OUTER JOIN Central WITH(NOLOCK) ON 
	Central.Panel_id = RealPanel.Realpanel_id 
LEFT OUTER JOIN TypeDevices WITH(NOLOCK) ON 
	TypeDevices.Device_id = central.Device_id 
LEFT OUTER JOIN Pults WITH(NOLOCK) ON 
	Pults.Id = RealPanel.Pult_id 
WHERE 
	COALESCE(Pults.IsOutPult, 0) = 0 
ORDER BY 
	RealPanel.panel_id, 
	groups.group_

SELECT
	panel.*,
	Engineers.engineer_name,
	Masters.MasterName,
	Installers.InstallerName,
	CONVERT(VARCHAR(10), CreateDate, 104) AS CD,
	central.panel_id AS central_id,
	mphone.panel_id AS mphone,
	stands.panel_id AS stand ,
	ISNULL(Groups.group_,-1) AS group_,
	Areas.Area_name,
	Pults.*,
	COALESCE(MPhone.LastTest, Central.LastTest) AS LastTest
FROM
	Panel WITH (NOLOCK)
LEFT OUTER JOIN Engineers WITH (NOLOCK) ON
	panel.engineer_id = engineers.engineer_id
LEFT OUTER JOIN Masters WITH (NOLOCK) ON
	panel.master_id = Masters.master_id
LEFT OUTER JOIN Installers WITH (NOLOCK) ON
	panel.installer_id = Installers.installer_id
LEFT JOIN Central WITH (NOLOCK) ON
	Central.panel_id = Panel.panel_id
LEFT OUTER JOIN groups WITH (NOLOCK) ON
	Panel.panel_id = Groups.panel_id
LEFT OUTER JOIN mphone WITH (NOLOCK) ON
	Panel.panel_id = mphone.panel_id
LEFT OUTER JOIN stands WITH (NOLOCK) ON
	((Panel.panel_id = stands.panel_id)
		AND(Stands.standorkey = 0))
LEFT OUTER JOIN Areas WITH (NOLOCK) ON
	Panel.Area_id = Areas.Area_id
LEFT OUTER JOIN Pults WITH (NOLOCK) ON
	Panel.Pult_id = Pults.id
WHERE
	panel.panel_id = 'L00001

	a.*,
	b.CompanyName,
	b.address,
	Telephones = b.Telephones,
	b.Director,
	b.Director2,
	b.Memo,
	b.TypeID,
	ct.Name AS TypeName,
	MoreAboutLun7.info_id,
	MoreAboutLun7.version,
	MoreAboutLun7.ExType_id,
	ExtensionTypes.Name AS ExtensionName,
	CONVERT(VARCHAR(5), Z8Timeout, 108) AS z8,
	CONVERT(VARCHAR(5), TimeForClose, 108) AS tfc ,
	engineers.engineer_name ,
	Customers.Organization,
	Customers.OrganizationAddress,
	Customers.Director AS CustomerDirector,
	ServiceOrganization.ServiceName,
	ISNULL(b.ServiceID,-1) AS ServiceID ,
	ISNULL(b.CustomerID,-1) AS CustomerID,
	NeedSendMessage = CASE
		WHEN EXISTS(SELECT * FROM dbo.Group_SMS_Code WHERE dbo.Group_SMS_Code.Panel_id = a.Panel_id AND dbo.Group_SMS_Code.Group_ = a.Group_) THEN 1
		ELSE 0
	END
FROM
	groups AS a WITH (NOLOCK)
INNER JOIN company AS b WITH (NOLOCK) ON
	a.CompanyID = b.ID
LEFT JOIN CompanyType AS ct WITH (NOLOCK) ON
	ct.ID = b.TypeID
LEFT JOIN MoreAboutLun7 WITH (NOLOCK) ON
	a.info_id = MoreAboutLun7.info_id
LEFT JOIN ExtensionTypes WITH (NOLOCK) ON
	MoreAboutLun7.ExType_id = ExtensionTypes.ExType_id
LEFT JOIN engineers WITH (NOLOCK) ON
	engineers.Work_Panel_id = a.Panel_id
LEFT JOIN vwRealPanel WITH (NOLOCK) ON
	vwRealPanel.Panel_id = a.panel_id
LEFT JOIN Pults WITH (NOLOCK) ON
	Pults.Id = vwRealPanel.Pult_Id
LEFT OUTER JOIN dbo.Customers ON
	b.CustomerID = dbo.Customers.ID
LEFT OUTER JOIN dbo.ServiceOrganization ON
	b.ServiceID = dbo.ServiceOrganization.ID
WHERE
	a.panel_id = 'L00028'
	AND group_ = 0

select 
vwArchives.Event_id, vwArchives.Panel_id,CONVERT(varchar,vwArchives.TimeEvent,3) as TEvent,
CONVERT(VARCHAR(10), vwArchives.TimeEvent , 108) as DEvent, vwArchives.Group_,
 CASE
WHEN vwArchives.Code in ('LIN','LOFF') THEN NULL
WHEN COALESCE(Groups.Message, '') <> '' THEN Groups.Message
WHEN COALESCE(GroupResponse.DESCRIPTION, '') <> '' THEN GroupResponse.DESCRIPTION
ELSE NULL END as GMessage,
TypeCode.Message, vwArchives.Event_Parent_id, vwArchives.Zone, TypeCode.idTCode, vwArchives.Line,
vwArchives.StateEvent,
vwArchives.Date_Key,
 CASE
WHEN Code.Code in ('Z34','Z35') THEN Code.Message
WHEN TypeCode.idTCode IN (1,2,14,105,126)
AND COALESCE(Zones.Message, '') <> '' THEN Zones.Message
WHEN TypeCode.idTCode IN (26,27,131) THEN Code.Message + COALESCE(SPACE(1) + Zones.Message, '')
WHEN COALESCE(vwArchives.Zone,0) = 0 THEN Code.Message + COALESCE(SPACE(1) + vwArchives.MeterCount, '')
WHEN TypeCode.idTCode IN (20,36,41,42) THEN Code.Message + COALESCE(SPACE(1) + vwArchives.MeterCount, '')
WHEN (Code.AccessCode = '1') and COALESCE(Users.UserName,'') <> '' THEN Users.UserName
WHEN (Code.AccessCode = '0') and COALESCE(Zones_GroupResponse.Message, '') <> '' THEN Zones_GroupResponse.Message
WHEN TypeCode.idTCode IN (151,152) THEN Code.Message + COALESCE(SPACE(1) + (SELECT LunOutput.OutputName FROM Mphone WITH(nolock)  inner join LunOutput WITH(nolock) on LunOutput.Mphone_id = Mphone.Mphone_id  WHERE Mphone.Panel_id = vwArchives.Panel_id  and LunOutput.OutputNum = vwArchives.Zone), '')
ELSE Code.Message + COALESCE(SPACE(1) + vwArchives.MeterCount, '') END as CodeMessage
,vwArchives.Code
,vwArchives.Phone, Code.FileName as CodeFileName, TypeCode.TypeCodeFileName
, CASE
WHEN vwArchives.BitMask & 2 = 2 THEN 5
WHEN COALESCE(Temp.StateEvent, 0) in (2,3) THEN 1
WHEN COALESCE(Panel.TestPanel, '0') = '1' THEN 2
WHEN COALESCE(Panel.Disabled, '0') = '1' THEN 3
ELSE 0 END as Object_Status
,CONVERT(VARCHAR,vwArchives.DeviceEventTime,3)+' '+CONVERT(VARCHAR,vwArchives.DeviceEventTime,108) as DeviceEventTime
, Company.address
, CASE
WHEN vwArchives.BitMask & 4 = 4 THEN 1
ELSE 0 END as UnknownObject
from vwArchives WITH(nolock)
LEFT JOIN Temp WITH(NOLOCK) ON temp.Event_id = vwArchives.Event_id
LEFT OUTER JOIN Groups WITH(nolock) ON vwArchives.Panel_id = Groups.Panel_id and vwArchives.Group_ = Groups.Group_
LEFT OUTER JOIN Code WITH(NOLOCK) ON Code.Code = vwArchives.Code and Code.CodeGroup = vwArchives.CodeGroup
LEFT OUTER JOIN TypeCode WITH(NOLOCK) ON TypeCode.idTCode = Code.idTCode
LEFT OUTER JOIN Users WITH(nolock) ON Users.Panel_id = vwArchives.Panel_id and Users.Group_ = vwArchives.Group_ and Users.UserCode = vwArchives.Zone
LEFT OUTER JOIN Zones WITH(nolock) ON Zones.Panel_id = vwArchives.Panel_id and Zones.Group_ = vwArchives.Group_ and Zones.Zone = vwArchives.Zone
LEFT OUTER JOIN GroupResponse WITH(NOLOCK) ON vwArchives.Panel_id = CAST(dbo.GroupResponse.Group_id AS VARCHAR(15))
LEFT OUTER JOIN vwRealPanel as Panel WITH(nolock) ON Panel.Panel_id = vwArchives.Panel_id
LEFT OUTER JOIN dbo.Company WITH(nolock) on Company.ID = Groups.companyID
LEFT OUTER JOIN dbo.Zones_GroupResponse WITH(NOLOCK) ON GroupResponse.Group_id = Zones_GroupResponse.Group_id AND
(vwArchives.Zone = Zones_GroupResponse.Zone AND  vwArchives.Panel_id = CAST(Zones_GroupResponse.Group_id AS VARCHAR(15)))
where vwArchives.Date_Key BETWEEN 20260319 and 20260401 and vwArchives.TimeEvent BETWEEN '20260319 00:00:00' and '20260401 23:59:59'  and (vwArchives.Panel_id LIKE 'L00028')
order by vwArchives.Event_id
