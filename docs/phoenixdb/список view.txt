--dbo.vwSMSQueue
CREATE   VIEW dbo.vwSMSQueue
AS
SELECT        CA.ID AS SMS_ID, CA.Result AS [Стан SMS], CA.Text AS [Текст SMS], CA.Recipient_ChannelNo AS Отримувач, CA.TimeEvent AS Час,
CA.Panel_id AS [Номер об'єкта], CA.Group_ AS Група, G.Message AS [Опис групи], CA.CodeGroup AS [Набір кодів], CA.Code AS [Код події],
C.Message AS [Опис коду], TC.Message AS [Тип коду], CA.Recipient_Type AS [Тип отримувача]
FROM            dbo.CurrentAlarms AS CA WITH (NOLOCK) LEFT OUTER JOIN
dbo.Group_SMS_Code AS SMS WITH (NOLOCK) ON SMS.Panel_id = CA.Panel_id AND SMS.Group_ = CA.Group_ AND SMS.Recipient_type = CA.Recipient_Type AND
SMS.Code = CA.Code AND SMS.CodeGroup = CA.CodeGroup LEFT OUTER JOIN
dbo.EventsRecipients AS ER WITH (NOLOCK) ON ER.Recipient_id = SMS.Recipient_id AND ER.RecipientIPorPhone = CA.Recipient_ChannelNo LEFT OUTER JOIN
dbo.Groups AS G WITH (NOLOCK) ON G.Panel_id = CA.Panel_id AND G.Group_ = CA.Group_ LEFT OUTER JOIN
dbo.Panel AS P WITH (NOLOCK) ON P.Panel_id = CA.Panel_id LEFT OUTER JOIN
dbo.Code AS C WITH (NOLOCK) ON C.Code = CA.Code AND C.CodeGroup = CA.CodeGroup LEFT OUTER JOIN
dbo.TypeCode AS TC WITH (NOLOCK) ON TC.idTCode = C.idTCode
"
